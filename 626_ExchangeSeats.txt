# Runtime: 264 ms, faster than 82.09% of MySQL online submissions for Exchange Seats.
SELECT
    CASE
        WHEN (id % 2 = 0) THEN id - 1
        WHEN (id % 2 = 1) AND id != (SELECT MAX(id) FROM Seat) THEN id + 1
        ELSE id
    END AS id,
    student
FROM Seat
ORDER BY id
