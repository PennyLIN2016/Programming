# Runtime: 1212 ms, faster than 5.01% of MySQL online submissions for Customer Placing the Largest Number of Orders.
SELECT
    t1.customer_number
FROM (
    SELECT customer_number, COUNT(*) AS OrderSum
    FROM orders
    GROUP BY customer_number
) AS t1
ORDER BY t1.OrderSum DESC
LIMIT 1
