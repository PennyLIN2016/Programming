# solution 1
# Runtime: 406 ms, faster than 48.20% of MySQL online submissions for Employees Earning More Than Their Managers.
SELECT
    T1.name AS Employee
 FROM
    Employee T1,
    Employee T2
 WHERE
    T1.managerId = T2.id
    AND T1.salary > T2.salary;

# solution 2
# Runtime: 547 ms, faster than 19.83% of MySQL online submissions for Employees Earning More Than Their Managers.
SELECT
    T1.name AS Employee
FROM
    Employee AS T1 LEFT JOIN Employee AS T2
ON
    T1.managerId = T2.id
WHERE
    T1.salary > T2.salary;
